   176→  const next = Math.min(1, cur + (speed * 1) / video.durationSec)
   177→  setVideoProgress(videoId, next)
   178→  accRef.current += speed
   179→  if (accRef.current > 60) { accRef.current = 0 }   // ← dead accumulator
   180→  if (next >= 1) setPlaying(false)
